### 1.2 textHighlights vs areaHighlights

> **参考**：Polar Bookshelf `PageMeta` — 明确区分两种高亮

```
textHighlights（文本高亮）
├─ 有 TextQuoteSelector（exact 文本）
├─ bbox = 文字选区坐标
├─ 可全文搜索
└─ 渲染：CSS 绝对定位半透明色块

areaHighlights（区域高亮）
├─ 无 TextQuoteSelector
├─ bbox = 矩形区域坐标
├─ 不可搜索
└─ 渲染：SVG/Canvas 矩形
```

## 二、Web 端渲染（React + PDF.js）

### 2.1 文本型标注 — CSS 绝对定位

```typescript
// web/src/components/AnnotationLayer.tsx

interface TextAnnotationProps {
  annotation: Annotation;
  pageWidth: number;
  pageHeight: number;
}

function TextAnnotation({ annotation, pageWidth, pageHeight }: TextAnnotationProps) {
  const { bbox, color, annotation_type } = annotation;
  const [x, y, w, h] = bbox;

  // 坐标转换为百分比（适配不同缩放级别）
  const style: React.CSSProperties = {
    position: 'absolute',
    left: `${(x / pageWidth) * 100}%`,
    top: `${(y / pageHeight) * 100}%`,
    width: `${(w / pageWidth) * 100}%`,
    height: `${(h / pageHeight) * 100}%`,
    pointerEvents: 'none',
  };

  switch (annotation_type) {
    case 'highlight':
      return <div style={{ ...style, backgroundColor: color, opacity: 0.3 }} />;
    case 'underline':
      return <div style={{ ...style, borderBottom: `2px solid ${color}` }} />;
    case 'squiggly':
      return <div style={{ ...style, borderBottom: `2px wavy ${color}` }} />;
    case 'strikethrough':
      return <div style={{ ...style, textDecoration: `line-through ${color}` }} />;
    default:
      return null;
  }
}
```

### 2.2 图形型标注 — SVG 渲染

```typescript
// web/src/components/ShapeAnnotation.tsx

function ShapeAnnotation({ annotation, pageWidth, pageHeight }: TextAnnotationProps) {
  const { stroke_data, color, annotation_type, opacity } = annotation;

  const svgStyle: React.CSSProperties = {
    position: 'absolute',
    top: 0,
    left: 0,
    width: '100%',
    height: '100%',
    pointerEvents: 'none',
  };

  switch (annotation_type) {
    case 'freehand':
      return (
        <svg style={svgStyle} viewBox={`0 0 ${pageWidth} ${pageHeight}`}>
          <path
            d={stroke_data.svg}  // 预计算的 SVG path
            stroke={color}
            strokeWidth={stroke_data.stroke_width || 2}
            fill="none"
            opacity={opacity}
          />
        </svg>
      );
    case 'circle':
      return (
        <svg style={svgStyle} viewBox={`0 0 ${pageWidth} ${pageHeight}`}>
          <circle
            cx={stroke_data.cx}
            cy={stroke_data.cy}
            r={stroke_data.r}
            stroke={color}
            strokeWidth={2}
            fill="none"
            opacity={opacity}
          />
        </svg>
      );
    case 'rectangle':
      return (
        <svg style={svgStyle} viewBox={`0 0 ${pageWidth} ${pageHeight}`}>
          <rect
            x={stroke_data.x}
            y={stroke_data.y}
            width={stroke_data.w}
            height={stroke_data.h}
            stroke={color}
            strokeWidth={2}
            fill="none"
            opacity={opacity}
          />
        </svg>
      );
    // ... line, polygon, etc.
    default:
      return null;
  }
}
```

## 三、Flutter 端渲染

### 3.1 文本型标注 — CustomPainter

```dart
// lib/features/reader/widgets/annotation_painter.dart

class AnnotationPainter extends CustomPainter {
  final List<Annotation> annotations;
  final double pageWidth;
  final double pageHeight;

  @override
  void paint(Canvas canvas, Size size) {
    final scaleX = size.width / pageWidth;
    final scaleY = size.height / pageHeight;

    for (final anno in annotations) {
      final paint = Paint()
        ..color = Color(int.parse(anno.color.replaceAll('#', '0xFF')))
        ..style = PaintingStyle.fill;

      final rect = Rect.fromLTWH(
        anno.bbox.x * scaleX,
        anno.bbox.y * scaleY,
        anno.bbox.w * scaleX,
        anno.bbox.h * scaleY,
      );

      switch (anno.annotationType) {
        case 'highlight':
          paint.alpha = (255 * 0.3).round();
          canvas.drawRect(rect, paint);
          break;
        case 'underline':
          paint.style = PaintingStyle.stroke;
          paint.strokeWidth = 2.0;
          canvas.drawLine(
            rect.bottomLeft, rect.bottomRight, paint);
          break;
        case 'freehand':
          _drawFreehand(canvas, anno.strokeData, scaleX, scaleY, paint);
          break;
        // ...
      }
    }
  }

  void _drawFreehand(Canvas canvas, StrokeData data,
      double scaleX, double scaleY, Paint paint) {
    paint.style = PaintingStyle.stroke;
    paint.strokeWidth = data.strokeWidth ?? 2.0;

    final path = Path();
    for (int i = 0; i < data.points.length; i++) {
      final point = data.points[i];
      final dx = point[0] * scaleX;
      final dy = point[1] * scaleY;
      if (i == 0) {
        path.moveTo(dx, dy);
      } else {
        path.lineTo(dx, dy);
      }
    }
    canvas.drawPath(path, paint);
  }
}
```

## 四、增量更新机制

> **参考**：Kotlin Toolkit `DecorationController` — `DecorationChange` 枚举 + `DiffUtil`

### 4.1 变更类型

```rust
// src/models/annotation.rs

/// 标注变更类型（用于增量更新）
#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(tag = "type", rename_all = "camelCase")]
pub enum AnnotationChange {
    /// 新增标注
    Added { annotation: Annotation },
    /// 更新标注（颜色/笔记/位置变化）
    Updated { annotation: Annotation },
    /// 移动标注（仅位置变化）
    Moved {
        id: String,
        from_page: u32,
        to_page: u32,
    },
    /// 删除标注
    Removed { id: String },
}
```

### 4.3 前端增量渲染

```typescript
// web/src/hooks/useAnnotations.ts

function useAnnotations(docId: number, page: number) {
  const [annotations, setAnnotations] = useState<Annotation[]>([]);
  const lastSync = useRef<string | null>(null);

  // 初始加载：全量
  useEffect(() => {
    fetchAnnotations(docId, page).then(data => {
      setAnnotations(data.items);
      lastSync.current = data.until;
    });
  }, [docId, page]);

  // 增量更新：SSE 或轮询
  useEffect(() => {
    const eventSource = new EventSource(
      `/api/v1/documents/${docId}/annotations/changes?since=${lastSync.current}`
    );

    eventSource.onmessage = (event) => {
      const changes: AnnotationChange[] = JSON.parse(event.data);
      setAnnotations(prev => applyChanges(prev, changes));
    };

    return () => eventSource.close();
  }, [docId]);
}

function applyChanges(
  current: Annotation[],
  changes: AnnotationChange[]
): Annotation[] {
  let result = [...current];

  for (const change of changes) {
    switch (change.type) {
      case 'added':
        result.push(change.annotation);
        break;
      case 'updated':
        result = result.map(a =>
          a.id === change.annotation.id ? change.annotation : a
        );
        break;
      case 'moved':
        // 页码变化，如果不在当前页则移除
        result = result.filter(a =>
          a.id !== change.id || a.page_num === change.to_page
        );
        break;
      case 'removed':
        result = result.filter(a => a.id !== change.id);
        break;
    }
  }

  return result;
}
```

## 五、标注的 `isActive` 状态

> **参考**：Kotlin Toolkit `Decoration.Style.Highlight.isActive`

标注有一个临时"激活态"，用于表示当前选中或正在编辑的标注：

```rust
/// 标注的激活状态（仅前端使用，不存储到数据库）
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AnnotationRenderState {
    /// 是否处于激活态（当前选中/编辑中）
    pub is_active: bool,
    /// 是否正在拖动
    pub is_dragging: bool,
    /// 是否正在编辑笔记
    pub is_editing: bool,
}
```

> **注意**：`isActive` 是前端渲染状态，不持久化到数据库。每次页面加载时重置为 `false`。

## 六、多列布局支持

> **参考**：Kotlin Toolkit `PagemarkType.SINGLE_COLUMN` + `column` 字段

对于多列 PDF 文档，标注需要携带列号信息：

```rust
/// 标注的列信息（多列布局）
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ColumnInfo {
    /// 列号（0-based，单列布局为 0）
    pub column: u32,
    /// 列总数
    pub total_columns: u32,
}
```

在 `annotations.metadata` JSONB 中存储：
```json
{
  "column": 1,
  "total_columns": 2
}
```

前端渲染时根据列号调整坐标偏移。
