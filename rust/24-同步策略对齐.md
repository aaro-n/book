// POST /api/v1/sync/annotations
async fn push_annotations(
    State(pool): State<PgPool>,
    Json(payload): Json<PushPayload>,
) -> Result<Json<PushResponse>> {
    let mut response = PushResponse {
        accepted: vec![],
        rejected: vec![],
    };
    
    for anno in payload.annotations {
        // 冲突检测：检查 _rev 是否匹配
        let conflict = check_conflict(&pool, &anno.id, &anno.rev).await?;
        
        if conflict {
            response.rejected.push(anno.id.clone());
            continue;
        }
        
        // 写入数据库
        insert_annotation(&pool, &anno).await?;
        response.accepted.push(anno.id.clone());
    }
    
    Ok(Json(response))
}

## 四、冲突解决策略

### 4.1 简单方案（MVP）

```
规则1: 最后修改者获胜（modified 时间戳比较）
规则2: 删除优先于修改（deleted=true 时，覆盖所有本地版本）
规则3: 同设备修改不会冲突（deviceId 相同时，直接覆盖）
```

### 4.2 实现逻辑

```rust
async fn resolve_conflict(
    pool: &PgPool,
    incoming: &SyncAnnotation,
    existing: &SyncAnnotation,
) -> ConflictResult {
    // 同设备 → 直接覆盖
    if incoming.device_id == existing.device_id {
        return ConflictResult::Accept;
    }
    
    // 删除优先
    if incoming.deleted == Some(true) {
        return ConflictResult::Accept;
    }
    
    // 最后修改者获胜
    let incoming_time = DateTime::parse_from_rfc3339(&incoming.modified.unwrap_or_default());
    let existing_time = DateTime::parse_from_rfc3339(&existing.modified.unwrap_or_default());
    
    if incoming_time > existing_time {
        ConflictResult::Accept
    } else {
        ConflictResult::Reject
    }
}
```

## 九、阅读进度同步的 LWW 冲突解决

> **参考**：Komga `markProgression` 方法 — 进度同步使用 LWW（Last Write Wins）策略。

### 9.1 冲突规则

| 规则 | 说明 |
|------|------|
| **规则1** | `modified` 时间戳更晚的获胜（LWW） |
| **规则2** | 同设备更新不冲突，直接覆盖（`device_id` 相同时） |
| **规则3** | 进度回退（如从 80% 跳回 20%）需要客户端确认 — 返回 `409 Conflict` + `hint` |
| **规则4** | `device` 信息必须随进度一起存储（`device_id` + `device_name`） |
| **规则5** | 删除优先于修改（`is_deleted=true` 时，覆盖所有版本） |

### 9.2 Rust 实现

```rust
// src/sync/conflict.rs

use chrono::{DateTime, Utc};

/// 进度同步冲突检测结果
pub enum ProgressionConflictResult {
    /// 接受新进度
    Accept,
    /// 拒绝：新进度比已存的旧
    RejectOlder { existing_modified: DateTime<Utc> },
    /// 拒绝：进度回退，需客户端确认
    RejectRegression { existing: f64, incoming: f64 },
    /// 接受：同设备直接覆盖
    AcceptSameDevice,
}

/// 进度冲突解决
pub fn resolve_progression_conflict(
    incoming_modified: DateTime<Utc>,
    incoming_progression: f64,
    incoming_device_id: &str,
    existing_modified: DateTime<Utc>,
    existing_progression: f64,
    existing_device_id: &str,
) -> ProgressionConflictResult {
    // 规则2：同设备 → 直接覆盖
    if incoming_device_id == existing_device_id {
        return ProgressionConflictResult::AcceptSameDevice;
    }

    // 规则1：LWW — 时间戳更晚的获胜
    if incoming_modified <= existing_modified {
        return ProgressionConflictResult::RejectOlder {
            existing_modified,
        };
    }

    // 规则3：进度回退检测（超过 20% 回退视为异常）
    let regression_threshold = 0.2;
    if existing_progression - incoming_progression > regression_threshold {
        return ProgressionConflictResult::RejectRegression {
            existing: existing_progression,
            incoming: incoming_progression,
        };
    }

    ProgressionConflictResult::Accept
}
```

## 十、EPUB 进度位置匹配与容错算法

> **参考**：Komga `analyzeAndPersist` — EPUB 进度使用三重定位 + 位置插值容错。

### 10.1 三重定位

EPUB 进度存储三个维度的位置信息，任一维度匹配即可恢复：

```rust
// src/models/progression.rs

pub struct EpubProgression {
    /// 维度1：资源 href（章节文件名）
    pub href: String,           // "chapter1.xhtml"

    /// 维度2：资源内进度 0.0 ~ 1.0
    pub progression: f64,      // 0.35

    /// 维度3：出版物全局位置索引（从 1 开始）
    pub position: u32,          // 15
}
```

### 10.2 内容变更后的匹配算法

当 EPUB 内容变更（章节增删）后，按以下顺序匹配：

```
1. href 精确匹配 → 成功 → 恢复进度
   失败 ↓

2. position 插值匹配 → 在新内容中找最接近的 position
   ├─ before.position < after.position → 插值成功
   └─ before.position > after.position → 位置矛盾，拒绝
   失败 ↓

3. progression 模糊匹配 → 找最接近的 progression 值
   失败 ↓

4. 重置到第 1 页（兜底）
```

### 10.3 Rust 实现

```rust
// src/sync/progression.rs

/// EPUB 进度匹配结果
pub enum ProgressionMatchResult {
    /// href 精确匹配
    ExactMatch { href: String, progression: f64 },
    /// position 插值匹配
    InterpolatedMatch { href: String, progression: f64, confidence: f64 },
    /// 匹配失败，重置
    Reset,
}

/// 在新的 EPUB 资源列表中匹配旧进度
pub fn match_epub_progression(
    old: &EpubProgression,
    new_resources: &[String], // 新的章节 href 列表
    new_total_positions: u32,
) -> ProgressionMatchResult {
    // 1. href 精确匹配
    if new_resources.contains(&old.href) {
        return ProgressionMatchResult::ExactMatch {
            href: old.href.clone(),
            progression: old.progression,
        };
    }

    // 2. position 插值
    if old.position > 0 && old.position <= new_total_positions {
        let ratio = old.position as f64 / new_total_positions as f64;
        let new_position = (ratio * new_total_positions as f64).round() as u32;
        let new_position = new_position.min(new_total_positions).max(1);
        let new_href = &new_resources[(new_position - 1) as usize];
        return ProgressionMatchResult::InterpolatedMatch {
            href: new_href.clone(),
            progression: old.progression,
            confidence: 0.7,
        };
    }

    // 3. progression 模糊匹配
    let target_position = (old.progression * new_total_positions as f64).round() as u32;
    let target_position = target_position.min(new_total_positions).max(1);
    if target_position > 0 {
        let new_href = &new_resources[(target_position - 1) as usize];
        return ProgressionMatchResult::InterpolatedMatch {
            href: new_href.clone(),
            progression: old.progression,
            confidence: 0.5,
        };
    }

    ProgressionMatchResult::Reset
}
```

## 十一、书籍内容变更后的进度迁移策略

> **参考**：Komga `analyzeAndPersist` — 页数变化时自动调整进度。

### 11.1 迁移规则

| 场景 | 处理方式 |
|------|---------|
| 页数增加 | 进度保持不变（页码不变） |
| 页数减少且当前页超出范围 | 重置到最后一页 |
| 页数减少且当前页在范围内 | 进度保持不变 |
| EPUB 章节增删 | 使用三重定位匹配（见第 10 节） |
| 标注位置失效 | 按 `TextQuoteSelector` 重新匹配，匹配失败标记 `position_changed=true` |

### 11.2 标注位置迁移

```rust
// src/service/migration_service.rs

/// 文档重新导入后，尝试迁移旧标注
pub async fn migrate_annotations(
    pool: &PgPool,
    doc_id: i32,
    old_page_count: u32,
    new_page_count: u32,
) -> Result<MigrationReport> {
    let mut report = MigrationReport::default();

    if old_page_count == new_page_count {
        return Ok(report); // 无需迁移
    }

    // 获取所有标注
    let annotations = get_annotations_by_doc(pool, doc_id).await?;

    for mut anno in annotations {
        let old_page = anno.page_num;

        // 页码超出范围 → 标记位置可能已变化
        if old_page > new_page_count {
            anno.page_num = new_page_count;
            anno.metadata["position_changed"] = json!(true);
            report.moved.push(anno.id);
            continue;
        }

        // 尝试用 TextQuoteSelector 重新匹配
        if let Some(Selector::TextQuoteSelector(tqs)) = extract_selector(&anno) {
            if let Some(new_page) = find_text_in_new_content(pool, doc_id, &tqs.exact).await? {
                anno.page_num = new_page;
                report.relocated.push(anno.id);
            } else {
                anno.metadata["position_changed"] = json!(true);
                report.unmatched.push(anno.id);
            }
        }

        update_annotation(pool, &anno).await?;
    }

    Ok(report)
}

pub struct MigrationReport {
    pub relocated: Vec<i32>,    // 成功重新匹配
    pub moved: Vec<i32>,        // 移到最后一页
    pub unmatched: Vec<i32>,    // 无法匹配，标记为位置已变化
}
```

## 十二、未来演进：CRDT 方案

> **参考**：`other/automerge-rs/` — 完整的 CRDT 同步协议。

### MVP 阶段：LWW 足够

个人书库场景下，标注的冲突主要是"创建/删除"冲突，LWW 策略已足够。

### Phase 2：多人协作时考虑 CRDT

如果未来增加分享/协作功能（多人同时编辑同一标注的注释），LWW 会导致后写者覆盖前写者的内容。此时可引入 automerge-rs：

| 场景 | MVP（LWW） | Phase 2（CRDT） |
|------|-----------|----------------|
| 进度同步 | ✅ LWW | 不需要 CRDT |
| 标注创建/删除 | ✅ LWW | 不需要 CRDT |
| 标注注释编辑 | ⚠️ LWW 覆盖 | ✅ CRDT 自动合并 |
| 多人协作笔记 | ❌ 不支持 | ✅ CRDT 实时协作 |

**automerge-rs 的优势**：
- 自动冲突解决，无需手动处理
- Bloom 过滤器优化同步数据量
- Sync State 可持久化，跨会话复用

**建议**：MVP 先用 LWW，在 `Cargo.toml` 中预留 `automerge = "0.5"` 依赖注释，Phase 2 再启用。
