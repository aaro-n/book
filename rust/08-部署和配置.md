## 🐳 Rust Dockerfile（多阶段构建）

```dockerfile
# ===== 构建阶段 =====
FROM rust:1.80-alpine AS builder

RUN apk add --no-cache musl-dev pkgconfig openssl-dev

WORKDIR /app
COPY Cargo.toml Cargo.lock ./
RUN mkdir src && echo "fn main() {}" > src/main.rs &&     cargo build --release &&     rm -rf src target/release/deps/pdf_manager*

COPY src/ src/
COPY config/ config/

RUN cargo build --release &&     strip target/release/pdf-manager

# ===== 运行阶段 =====
FROM alpine:3.20

RUN apk add --no-cache ca-certificates tzdata

COPY --from=builder /app/target/release/pdf-manager /usr/local/bin/
COPY --from=builder /app/config/ /etc/pdf-manager/

EXPOSE 8080
CMD ["pdf-manager"]
```

### 镜像大小对比

| 阶段 | Go (Alpine) | Rust (Alpine) |
|------|------------|---------------|
| 构建镜像 | ~600MB | ~1.2GB（首次编译） |
| 运行镜像 | ~15-25MB | ~10-18MB |
| Scratch 镜像 | ~8MB | ~5MB（musl 静态链接） |

### docker-compose.yml（完整版）

```yaml
version: "3.8"

services:
  # PostgreSQL 18 - 主数据库
  postgres:
    image: postgres:18-alpine
    container_name: pdf-manager-db
    environment:
      POSTGRES_USER: pdfmanager
      POSTGRES_PASSWORD: ${DB_PASSWORD:-change_me}
      POSTGRES_DB: pdf_manager
      POSTGRES_INITDB_ARGS: "--encoding=UTF8 --locale=en_US.UTF-8"
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./internal/db/migrations:/docker-entrypoint-initdb.d
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U pdfmanager"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - pdf-network

  # Redis - 缓存（可选）
  redis:
    image: redis:7-alpine
    container_name: pdf-manager-cache
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - pdf-network

  # Elasticsearch - 搜索引擎（推荐，含IK分词器）
  elasticsearch:
    image: docker.elastic.co/elasticsearch/elasticsearch:8.10.0
    container_name: pdf-manager-search
    environment:
      - discovery.type=single-node
      - xpack.security.enabled=false
      - ES_JAVA_OPTS=-Xms512m -Xmx512m
    ports:
      - "9200:9200"
    volumes:
      - elasticsearch_data:/usr/share/elasticsearch/data
    # 安装IK分词器的脚本
    command: >
      /bin/sh -c "./bin/elasticsearch-plugin list | grep analysis-ik || ./bin/elasticsearch-plugin install https://github.com/medcl/elasticsearch-analysis-ik/releases/download/v8.10.0/elasticsearch-analysis-ik-8.10.0.zip;
      /usr/local/bin/docker-entrypoint.sh elasticsearch"
    healthcheck:
      test: ["CMD-SHELL", "curl -f http://localhost:9200 || exit 1"]
      interval: 10s
      timeout: 30s
      retries: 5
    networks:
      - pdf-network

  # Meilisearch - 轻量搜索引擎（备选）
  meilisearch:
    image: getmeili/meilisearch:v1.7.0
    container_name: pdf-manager-meilisearch
    environment:
      MEILI_NO_ANALYTICS: "true"
      MEILI_MASTER_KEY: "mastersecret"
    ports:
      - "7700:7700"
    volumes:
      - meilisearch_data:/meili_data
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:7700/health"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - pdf-network

  # Typesense - 高速轻量搜索引擎（备选）🆕
  typesense:
    image: typesense/typesense:27.0
    container_name: pdf-manager-typesense
    environment:
      TYPESENSE_API_KEY: typesense_secret
      TYPESENSE_DATA_DIR: /data
      TYPESENSE_ENABLE_CORS: "true"
    ports:
      - "8108:8108"
    volumes:
      - typesense_data:/data
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8108/health"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - pdf-network

  # Sonic - 极致省内存搜索引擎（极简模式）🆕
  sonic:
    image: valeriansaliou/sonic:v1.4.9
    container_name: pdf-manager-sonic
    environment:
      SONIC_SEARCH_PASSWORD: SecretPassword
      SONIC_INGEST_PASSWORD: SecretPassword
    volumes:
      - sonic_data:/var/lib/sonic/store
      - ./config/sonic.cfg:/etc/sonic.cfg:ro
    networks:
      - pdf-network

  # MinIO - S3 Compatible Storage
  minio:
    image: minio/minio:RELEASE.2024-01-18T22-51-28Z
    container_name: pdf-manager-minio
    ports:
      - "9000:9000"
      - "9001:9001"
    environment:
      MINIO_ROOT_USER: minioadmin
      MINIO_ROOT_PASSWORD: minioadmin
    command: server /data --console-address ":9001"
    volumes:
      - minio_data:/data
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 30s
      timeout: 20s
      retries: 3
    networks:
      - pdf-network

  # MinIO Bucket Setup
  createbuckets:
    image: minio/mc:RELEASE.2024-01-16T16-06-34Z
    depends_on:
      minio:
        condition: service_healthy
    entrypoint: >
      /bin/sh -c "
      /usr/bin/mc alias set myminio http://minio:9000 minioadmin minioadmin;
      /usr/bin/mc mb myminio/documents;
      /usr/bin/mc policy set public myminio/documents;
      exit 0;
      "
    networks:
      - pdf-network

  # Rust应用
  app:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: pdf-manager-app
    environment:
      # 数据库
      DB_HOST: postgres
      DB_PORT: 5432
      DB_USER: pdfmanager
      DB_PASSWORD: ${DB_PASSWORD:-change_me}
      DB_NAME: pdf_manager
      # S3存储 (MinIO)
      S3_ENDPOINT: http://minio:9000
      S3_REGION: us-east-1
      S3_BUCKET: documents
      S3_ACCESS_KEY: minioadmin
      S3_SECRET_KEY: minioadmin
      S3_USE_PATH_STYLE: "true"
      # Redis
      REDIS_ADDR: redis:6379
      # Search Engine（Set SEARCH_ENGINE_TYPE to: elasticsearch|meilisearch|typesense|sonic|postgres）
      SEARCH_ENGINE_TYPE: ${SEARCH_ENGINE_TYPE:-elasticsearch}
      ELASTICSEARCH_URL: http://elasticsearch:9200
      MEILISEARCH_URL: http://meilisearch:7700
      MEILISEARCH_KEY: mastersecret
      TYPESENSE_URL: http://typesense:8108
      TYPESENSE_API_KEY: typesense_secret
      SONIC_HOST: sonic
      SONIC_SEARCH_PORT: 1491
      SONIC_INGEST_PORT: 1492
      SONIC_PASSWORD: SecretPassword
    ports:
      - "8080:8080"
    depends_on:
      postgres:
        condition: service_healthy
      minio:
        condition: service_healthy
    networks:
      - pdf-network

volumes:
  postgres_data:
  redis_data:
  elasticsearch_data:
  meilisearch_data:
  typesense_data:
  sonic_data:
  minio_data:

networks:
  pdf-network:
    driver: bridge
```

### 配置文件 (config/app.yaml)

```yaml
app:
  name: "pdf-manager"
  env: "dev"
  port: 8080
  log_level: "debug"
  jwt_secret: "your-256-bit-secret"

database:
  dsn: "postgres://pdfmanager:change_me@postgres:5432/pdf_manager?sslmode=disable"
  # sqlx 连接池配置
  max_connections: 25
  min_connections: 5

redis:
  addr: "redis:6379"
  password: ""
  db: 0

s3:
  endpoint: "http://minio:9000"
  region: "us-east-1"
  bucket: "documents"
  access_key: "minioadmin"
  secret_key: "minioadmin"
  use_path_style: true

# 搜索引擎设置
search_engine:
  # type: elasticsearch | meilisearch | typesense | sonic | postgres
  type: "elasticsearch"
  elasticsearch:
    url: "http://elasticsearch:9200"
    index_prefix: "pdf_"
  meilisearch:
    url: "http://meilisearch:7700"
    master_key: "mastersecret"
  typesense:
    url: "http://typesense:8108"
    api_key: "typesense_secret"
  sonic:
    host: "sonic"
    search_port: 1491
    ingest_port: 1492
    password: "SecretPassword"

# 分词器 (仅在 use_search_engine="postgres" 时生效)
segmenter:
  # mode: none | gse | jieba | mixed
  # "none" when using ES/Meili/Typesense (built-in CJK)
  # "gse" recommended with Sonic (Sonic doesn't do CJK)
  # "mixed" recommended with PG Only
  mode: "none"
  dict_path: "config/dictionary.txt"
  mixed_strategy: "deferred"  # concurrent | deferred

jwt:
  secret: "your-256-bit-secret"
  expiration_hours: 24
```

### 环境变量说明

| 变量名 | 默认值 | 说明 |
|--------|--------|------|
| DB_PASSWORD | change_me | 数据库密码 |
| MINIO_ROOT_PASSWORD | minioadmin | MinIO管理员密码 |
| MEILI_MASTER_KEY | mastersecret | Meilisearch主密钥 |
